15   def get_links(VIM, gene_names, regulators, sort=True, file_name=None):
16       idx = [i for i, gene in enumerate(gene_names) if gene in regulators]
17   
18       # Create a list of predicted edges
19       pred_edges = [(gene_names[j], gene_names[i], score)
20                     for (i, j), score in np.ndenumerate(VIM) if i != j and j in idx]
21   
22       # Convert to DataFrame
23       pred_edges = pd.DataFrame(pred_edges, columns=['Gene1', 'Gene2', 'Importance'])
24   
25       # Sort the DataFrame by Importance if required
26       if sort:
27           pred_edges.sort_values(by='Importance', ascending=False, inplace=True)
28   
29       # save to file, otherwise return the DataFrame
30       if file_name is None:
31           return pred_edges
32       else:
33           pred_edges.to_csv(file_name, sep='\t', header=None, index=None)
34           return None  # Avoid returning a DataFrame if saving to a file

37   def get_importances(TS_data, time_points, time_lag, gene_names, regulators, alpha, SS_data=None, param={}):
38       time_start = time.time()
39   
40       ngenes = TS_data[0].shape[1]
41   
42       alphas = [alpha] * ngenes
43   
44       # Get the indices of the candidate regulators
45       idx = [i for i, gene in enumerate(gene_names) if gene in regulators]
46   
47       # Learn an ensemble of trees for each target gene, and compute scores for candidate regulators
48       VIM = np.zeros((ngenes, ngenes))
49   
50       for i in range(ngenes):
51           input_idx = idx.copy()
52           if i in input_idx:
53               input_idx.remove(i)
54           vi = get_importances_single(TS_data, time_points, time_lag, alphas[i], input_idx, i, SS_data, param)
55           VIM[i, :] = vi
56   
57       time_end = time.time()
58       print("Elapsed time: %.2f seconds" % (time_end - time_start))
59   
60       return VIM

63   def get_importances_single(TS_data, time_points, time_lag, alpha, input_idx, output_idx, SS_data, param):
64       h = 1  # lag (in number of time points) used for the finite approximation of the derivative of the target gene expression
65       ngenes = TS_data[0].shape[1]
66       nexp = len(TS_data)
67       nsamples_time = sum([expr_data.shape[0] for expr_data in TS_data])
68       ninputs = len(input_idx)
69       # Construct learning sample
70       # Time-series data
71       input_matrix_time = np.zeros((nsamples_time - h * nexp, ninputs))
72       output_vect_time = np.zeros(nsamples_time - h * nexp)
73       nsamples_count = 0
74       for (i, current_timeseries) in enumerate(TS_data):
75           current_time_points = time_points[i]
76           npoints = current_timeseries.shape[0]
77           time_diff_current = current_time_points[h:] - current_time_points[:npoints - h]
78           current_timeseries_input = current_timeseries[:npoints - h, input_idx]
79           current_timeseries_output = (current_timeseries[h:, output_idx] - current_timeseries[:npoints - h,
80                                                                             output_idx]) / time_diff_current + alpha * current_timeseries[
81                                                                                                                        :npoints - h,
82                                                                                                                        output_idx]
83   
84           # Time delay
85           npoints = current_timeseries_input.shape[0]
86           current_timeseries_input = current_timeseries_input[:npoints - time_lag]
87           current_timeseries_output = current_timeseries_output[time_lag:]
88   
89           nsamples_current = current_timeseries_input.shape[0]
90           input_matrix_time[nsamples_count:nsamples_count + nsamples_current, :] = current_timeseries_input
91           output_vect_time[nsamples_count:nsamples_count + nsamples_current] = current_timeseries_output
92           nsamples_count += nsamples_current
93   
94       # Steady-state data
95       if SS_data is not None:
96           input_matrix_steady = SS_data[:, input_idx]
97           output_vect_steady = SS_data[:, output_idx] * alpha
98   
99           # Concatenation
100          input_all = np.vstack([input_matrix_steady, input_matrix_time])
101          output_all = np.concatenate((output_vect_steady, output_vect_time))
102      else:
103          input_all = input_matrix_time
104          output_all = output_vect_time
105  
106      treeEstimator = XGBRegressor(**param)
107  
108      # Learn ensemble of trees
109      treeEstimator.fit(input_all, output_all)
110  
111      # Compute importance scores
112      feature_importances = treeEstimator.feature_importances_
113      vi = np.zeros(ngenes)
114      vi[input_idx] = feature_importances
115  
116      return vi

119  def get_scores(VIM, gold_edges, gene_names, regulators):
120      idx = [i for i, gene in enumerate(gene_names) if gene in regulators]
121      pred_edges = [(gene_names[j], gene_names[i], score) for (i, j), score in np.ndenumerate(VIM) if i != j and j in idx]
122      pred_edges = pd.DataFrame(pred_edges)
123      # pred_edges = pred_edges.iloc[:100000]
124      final = pd.merge(pred_edges, gold_edges, on=[0, 1], how='inner')
125      auroc = roc_auc_score(final['2_y'], final['2_x'])
126      aupr = average_precision_score(final['2_y'], final['2_x'])
127      return auroc, aupr

253  def evaluate(individual, TS_data, time_points, time_lag, gene_names, regulators, SS_data, gold_edges):
254      """ 
255      Evaluate the fitness of an individual based on the AUC score and AP score. 
256      The individual represents a set of hyperparameters for XGBoost. 
257      """
258      # Unpack individual hyperparameters
259      n_estimators, learning_rate, max_depth = individual
260  
261      # Define the XGBoost parameters
262      xgb_kwargs = {
263          'n_estimators': int(n_estimators),
264          'learning_rate': learning_rate,
265          'max_depth': int(max_depth),
266          'importance_type': "weight",
267          'n_jobs': -1,
268          'objective': 'reg:squarederror'
269      }
270  
271      # Get the importance values using the current hyperparameters
272      VIM = get_importances(TS_data, time_points, time_lag, gene_names, regulators, alpha=0.022408670532763,
273                            SS_data=SS_data, param=xgb_kwargs)
274  
275      # Calculate AUROC and AUPR scores
276      auroc, aupr = get_scores(VIM, gold_edges, gene_names, regulators)
277  
278      # Return a tuple as expected by DEAP (fitness values)
279      return auroc, aupr

282  # Genetic algorithm setup with elitism
283  creator.create("FitnessMax", base.Fitness, weights=(1.0, 1.0))  # Maximize both AUROC and AUPR
284  creator.create("Individual", list, fitness=creator.FitnessMax)
285  
286  toolbox = base.Toolbox()
287  
288  # Define the bounds for the hyperparameters
289  BOUNDS_LOW = [50, 0.001, 3]  # n_estimators: [50, 500], learning_rate: [0.001, 0.1], max_depth: [3, 15]
290  BOUNDS_HIGH = [500, 0.1, 15]
291  
292  # Register the creation of individuals with random values within bounds
293  for i in range(3):  # We have 3 hyperparameters
294      toolbox.register(f"hyperparameter_{i}", random.uniform, BOUNDS_LOW[i], BOUNDS_HIGH[i])
295  
296  # Create individuals based on the hyperparameter bounds
297  toolbox.register("individualCreator", tools.initCycle, creator.Individual,
298                   tuple([toolbox.__getattribute__(f"hyperparameter_{i}") for i in range(3)]), n=1)
299  
300  toolbox.register("population", tools.initRepeat, list, toolbox.individualCreator)
301  
302  # Register crossover, mutation, and selection methods
303  toolbox.register("mate", tools.cxSimulatedBinaryBounded, low=BOUNDS_LOW, up=BOUNDS_HIGH, eta=1.0)
304  toolbox.register("mutate", tools.mutPolynomialBounded, low=BOUNDS_LOW, up=BOUNDS_HIGH, eta=1.0, indpb=1.0/3)
305  toolbox.register("select", tools.selTournament, tournsize=3)
306  
307  toolbox.register("evaluate", evaluate, TS_data=TS_data, time_points=time_points, time_lag=0, gene_names=gene_names,
308                   regulators=regulators, SS_data=SS_data, gold_edges=gold_edges)
309  
310  # Set GA parameters
311  population_size = 30
312  generations = 10
313  cx_probability = 0.7
314  mut_probability = 0.3
315  
316  # Create an initial population
317  population = toolbox.population(n=population_size)
318  
319  # Hall of Fame for elitism
320  hof = tools.HallOfFame(1)
321  
322  # Define the statistics (optional)
323  stats = tools.Statistics(lambda ind: ind.fitness.values)
324  stats.register("avg", np.mean)
325  stats.register("std", np.std)
326  stats.register("min", np.min)
327  stats.register("max", np.max)
328  
329  # Run the genetic algorithm with elitism using eaSimple
330  for gen in range(generations):
331      print(f"Generation {gen}")
332  
333      # Evaluate all individuals in the population
334      fitnesses = list(map(toolbox.evaluate, population))
335      for ind, fit in zip(population, fitnesses):
336          ind.fitness.values = fit
337  
338      # Select the best individuals for mating
339      selected = toolbox.select(population, len(population))
340      offspring = list(map(toolbox.clone, selected))
341  
342      # Apply crossover and mutation
343      for child1, child2 in zip(offspring[::2], offspring[1::2]):
344          if random.random() < cx_probability:
345              toolbox.mate(child1, child2)
346              del child1.fitness.values
347              del child2.fitness.values
348  
349      for mutant in offspring:
350          if random.random() < mut_probability:
351              toolbox.mutate(mutant)
352              del mutant.fitness.values
353  
354      # Evaluate the fitness of the new offspring
355      for ind in offspring:
356          if not ind.fitness.valid:
357              ind.fitness.values = toolbox.evaluate(ind)
358  
359      # Update the Hall of Fame
360      hof.update(offspring)
361  
362      # Replace the old population with the new one
363      population[:] = offspring
364  
365      # Log statistics
366      record = stats.compile(population)
367      print(record)
368  
369  # After running GA, output the best individual
370  best_individual = hof[0]
371  print(f"Best individual: {best_individual}")
372  print(f"Fitness: {best_individual.fitness.values}")

377  def plot_network(predicted_edges, gene_names):
378      """ 
379      Visualizes the gene regulatory network (GRN) based on predicted edges. 
380      :param predicted_edges: DataFrame with 'Gene1', 'Gene2', and 'Importance' columns 
381      :param gene_names: List of gene names 
382      """
383  
384      # check if predicted_edges is not None
385      if predicted_edges is not None:
386          # filter predicted_edges for importance > 0.25
387          filtered_edges = predicted_edges[predicted_edges['Importance'] > 0.25]
388  
389          # Add edges with weights (Importance) from predicted edges
390          for index, row in filtered_edges.iterrows():
391              print(f"Gene1: {row['Gene1']}, Gene2: {row['Gene2']}, Importance: {row['Importance']}")
392          G = nx.DiGraph()
393  
394              #G.add_edge(row['Gene1'], row['Gene2'], weight=row['Importance'])
395          for index, row in filtered_edges.iterrows():
396              G.add_edge(row['Gene1'], row['Gene2'], weight=row['Importance'])
397  
398          # Create a directed graph
399  
400          # Add nodes for each gene
401          G.add_nodes_from(gene_names)
402  
403  
404          # Set up the plot
405          plt.figure(figsize=(12, 12))
406          pos = nx.spring_layout(G, k=0.15, iterations=20)  # Use spring layout for node placement
407  
408          # Draw nodes and edges
409          # nx.draw(G, pos, with_labels=True, node_size=3000, node_color='skyblue', font_size=10, font_weight='bold',edge_color='gray', width=1, alpha=0.7)
410          nx.draw(G, pos, with_labels=True, node_size=1000, node_color='skyblue', font_size=10, font_weight='bold',
411                  arrows=True)
412  
413          # Draw edge weights (importance values)
414          edge_labels = nx.get_edge_attributes(G, 'weight')
415          nx.draw_networkx_edge_labels(G, pos, edge_labels=edge_labels)
416  
417          # Show the plot
418          plt.title("Predicted Gene Regulatory Network (GRN)")
419          plt.show()
